grilops.paths

This module supports puzzles where paths are filled into the grid.

These paths may be either closed (loops) or open ("terminated" paths).

 19class PathSymbolSet(SymbolSet):
 20  """A `grilops.symbols.SymbolSet` consisting of symbols that may form paths.
 21
 22  Additional symbols (e.g. a `grilops.symbols.Symbol` representing an empty
 23  space) may be added to this `grilops.symbols.SymbolSet` by calling
 24  `grilops.symbols.SymbolSet.append` after it's constructed.
 25
 26  Args:
 27    lattice (grilops.geometry.Lattice): The structure of the grid.
 28    include_terminals (bool): If True, create symbols for path terminals.
 29      Defaults to True.
 30  """
 31
 32  def __init__(self, lattice: Lattice, include_terminals: bool = True):
 33    super().__init__([])
 34    self.__include_terminals = include_terminals
 35
 36    self.__symbols_for_direction: Dict[Direction, List[int]] = defaultdict(list)
 37    self.__symbol_for_direction_pair: Dict[Tuple[Direction, Direction], int] = {}
 38    self.__terminal_for_direction: Dict[Direction, int] = {}
 39
 40    dirs = lattice.edge_sharing_directions()
 41
 42    for idx, (di, dj) in enumerate(itertools.combinations(dirs, 2)):
 43      lbl = lattice.label_for_direction_pair(di, dj)
 44      self.append(di.name + dj.name, lbl)
 45      self.__symbols_for_direction[di].append(idx)
 46      self.__symbols_for_direction[dj].append(idx)
 47      self.__symbol_for_direction_pair[(di, dj)] = idx
 48      self.__symbol_for_direction_pair[(dj, di)] = idx
 49      self.__max_path_segment_symbol_index = idx
 50
 51    if include_terminals:
 52      for d in dirs:
 53        self.append(d.name, lattice.label_for_direction(d))
 54        idx = self.max_index()
 55        self.__symbols_for_direction[d].append(idx)
 56        self.__terminal_for_direction[d] = idx
 57        self.__max_path_terminal_symbol_index = idx
 58
 59  def is_path(self, symbol: ArithRef) -> BoolRef:
 60    """Returns true if the given symbol represents part of a path.
 61
 62    Args:
 63      symbol (ArithRef): An `ArithRef` expression representing a symbol.
 64
 65    Returns:
 66      A true `BoolRef` if the symbol represents part of a path.
 67    """
 68    if self.__include_terminals:
 69      return symbol < self.__max_path_terminal_symbol_index + 1
 70    return symbol < self.__max_path_segment_symbol_index + 1
 71
 72  def is_path_segment(self, symbol: ArithRef) -> BoolRef:
 73    """Returns true if the given symbol represents a non-terminal path segment.
 74
 75    Args:
 76      symbol (ArithRef): An `ArithRef` expression representing a symbol.
 77
 78    Returns:
 79      A true `BoolRef` if the symbol represents a non-terminal path segment.
 80    """
 81    return symbol < self.__max_path_segment_symbol_index + 1
 82
 83  def is_terminal(self, symbol: ArithRef) -> BoolRef:
 84    """Returns true if the given symbol represents a path terminal.
 85
 86    Args:
 87      symbol (ArithRef): An `ArithRef` expression representing a symbol.
 88
 89    Returns:
 90      A true `BoolRef` if the symbol represents a path terminal
 91    """
 92    if not self.__include_terminals:
 93      return BoolVal(False)
 94    return And(
 95      symbol > self.__max_path_segment_symbol_index,
 96      symbol < self.__max_path_terminal_symbol_index + 1
 97    )

112    """Returns the symbol with arms going in the two given directions.

137class PathConstrainer:
138  """Creates constraints for ensuring symbols form connected paths.
139
140  Args:
141    symbol_grid (grilops.grids.SymbolGrid): The grid to constrain.
142    complete (bool): If True, every cell must be part of a path.
143      Defaults to False.
144    allow_terminated_paths (bool): If True, finds paths that are terminated
145      (not loops). Defaults to True.
146    allow_loops (bool): If True, finds paths that are loops. Defaults to True.
147  """
148  _instance_index = 0
149
150  def __init__(
151      self,
152      symbol_grid: SymbolGrid,
153      complete: bool = False,
154      allow_terminated_paths: bool = True,
155      allow_loops: bool = True,
156  ):
157    PathConstrainer._instance_index += 1
158
159    self.__symbol_grid = symbol_grid
160    self.__complete = complete
161    self.__allow_terminated_paths = allow_terminated_paths
162    self.__allow_loops = allow_loops
163    self.__num_paths: Optional[ArithRef] = None
164
165    self.__path_instance_grid: Dict[Point, ArithRef] = {
166      p: Int(f"pcpi-{PathConstrainer._instance_index}-{p.y}-{p.x}")
167      for p in self.__symbol_grid.grid.keys()
168    }
169    self.__path_order_grid: Dict[Point, ArithRef] = {
170      p: Int(f"pcpo-{PathConstrainer._instance_index}-{p.y}-{p.x}")
171      for p in self.__symbol_grid.grid.keys()
172    }
173
174    self.__add_path_edge_constraints()
175    self.__add_path_instance_grid_constraints()
176    self.__add_path_order_grid_constraints()
177    self.__add_allow_terminated_paths_constraints()
178
179  def __add_path_edge_constraints(self):
180    solver = self.__symbol_grid.solver
181    sym: PathSymbolSet = cast(PathSymbolSet, self.__symbol_grid.symbol_set)
182
183    for p, cell in self.__symbol_grid.grid.items():
184      for d in self.__symbol_grid.lattice.edge_sharing_directions():
185        np = p.translate(d.vector)
186        dir_syms = sym.symbols_for_direction(d)
187        ncell = self.__symbol_grid.grid.get(np, None)
188        if ncell is not None:
189          opposite_syms = sym.symbols_for_direction(
190            self.__symbol_grid.lattice.opposite_direction(d))
191          cell_points_dir = Or(*[cell == s for s in dir_syms])
192          neighbor_points_opposite = Or(*[ncell == s for s in opposite_syms])
193          solver.add(Implies(cell_points_dir, neighbor_points_opposite))
194        else:
195          for s in dir_syms:
196            solver.add(cell != s)
197
198  def __add_path_instance_grid_constraints(self):
199    solver = self.__symbol_grid.solver
200    sym: PathSymbolSet = cast(PathSymbolSet, self.__symbol_grid.symbol_set)
201
202    for p, pi in self.__path_instance_grid.items():
203      if self.__complete:
204        solver.add(pi >= 0)
205      else:
206        solver.add(pi >= -1)
207      solver.add(pi < len(self.__symbol_grid.grid))
208
209      cell = self.__symbol_grid.grid[p]
210      solver.add(sym.is_path(cell) == (pi != -1))
211      point_index = self.__symbol_grid.lattice.point_to_index(p)
212      assert point_index is not None
213      solver.add(
214        (self.__path_order_grid[p] == 0) ==
215        (pi == point_index)
216      )
217      for d in self.__symbol_grid.lattice.edge_sharing_directions():
218        dir_syms = sym.symbols_for_direction(d)
219        np = p.translate(d.vector)
220        ncell = self.__symbol_grid.grid.get(np, None)
221        if ncell is not None:
222          cell_points_dir = Or(*[cell == s for s in dir_syms])
223          solver.add(
224            Implies(
225              cell_points_dir,
226              pi == self.__path_instance_grid[np]
227            )
228          )
229
230  def __all_direction_pairs(self) -> Iterable[Tuple[int, Direction, Direction]]:
231    dirs = self.__symbol_grid.lattice.edge_sharing_directions()
232    for idx, (di, dj) in enumerate(itertools.combinations(dirs, 2)):
233      yield (idx, di, dj)
234
235  def __add_path_order_grid_constraints(self):
236    solver = self.__symbol_grid.solver
237    sym: PathSymbolSet = cast(PathSymbolSet, self.__symbol_grid.symbol_set)
238
239    for p, po in self.__path_order_grid.items():
240      if self.__complete:
241        solver.add(po >= 0)
242      else:
243        solver.add(po >= -1)
244
245      cell = self.__symbol_grid.grid[p]
246      solver.add(sym.is_path(cell) == (po != -1))
247
248      for d in self.__symbol_grid.lattice.edge_sharing_directions():
249        try:
250          s = sym.terminal_for_direction(d)
251        except KeyError:
252          continue
253        np = p.translate(d.vector)
254        if np in self.__path_order_grid:
255          solver.add(Implies(
256            cell == s,
257            Or(
258              And(
259                self.__path_order_grid[p] == 0,
260                self.__path_order_grid[np] == 1
261              ),
262              And(
263                self.__path_order_grid[p] > 0,
264                self.__path_order_grid[np] == self.__path_order_grid[p] - 1
265              )
266            )
267          ))
268
269      for idx, d1, d2 in self.__all_direction_pairs():
270        pi = p.translate(d1.vector)
271        pj = p.translate(d2.vector)
272        if pi in self.__path_order_grid and pj in self.__path_order_grid:
273          solver.add(Implies(
274            cell == idx,
275            self.__path_order_grid[pi] != self.__path_order_grid[pj]
276          ))
277          solver.add(Implies(
278            And(cell == idx, po > 0),
279            Or(
280              And(
281                self.__path_order_grid[pi] == po - 1,
282                Or(
283                  self.__path_order_grid[pj] == po + 1,
284                  self.__path_order_grid[pj] == 0 if self.__allow_loops else BoolVal(False)
285                )
286              ),
287              And(
288                Or(
289                  self.__path_order_grid[pi] == po + 1,
290                  self.__path_order_grid[pi] == 0 if self.__allow_loops else BoolVal(False)
291                ),
292                self.__path_order_grid[pj] == po - 1
293              ),
294            )
295          ))
296
297  def __add_allow_terminated_paths_constraints(self):
298    if not self.__allow_terminated_paths:
299      for cell in self.__symbol_grid.grid.values():
300        self.__symbol_grid.solver.add(
301          Not(self.__symbol_grid.symbol_set.is_terminal(cell)))
302
303  @property
304  def num_paths(self) -> ArithRef:
305    """A constant representing the number of distinct paths found."""
306    if self.__num_paths is None:
307      self.__num_paths = Sum(*[
308        If(self.__path_order_grid[p] == 0, 1, 0)
309        for p in self.__symbol_grid.lattice.points
310      ])
311    return self.__num_paths
312
313  @property
314  def path_instance_grid(self) -> Dict[Point, ArithRef]:
315    """Constants of path instance identification.
316
317    Each separate path will have a distinct instance number. The instance number
318    is -1 if the cell does not contain a path segment or terminal.
319    """
320    return self.__path_instance_grid
321
322  @property
323  def path_order_grid(self) -> Dict[Point, ArithRef]:
324    """Constants of path traversal orders.
325
326    Each segment or terminal of a path will have a distinct order number. The
327    order number is -1 if the cell does not contain a path segment or terminal.
328    """
329    return self.__path_order_grid
330
331  def print_path_numbering(self):
332    """Prints the path instance and order for each path cell.
333
334    Should be called only after the solver has been checked.
335    """
336    model = self.__symbol_grid.solver.model()
337    def print_function(p):
338      pi = model.eval(self.__path_instance_grid[p]).as_long()
339      po = model.eval(self.__path_order_grid[p]).as_long()
340      if pi == -1:
341        return "    "
342      return f"{chr(pi+65)}{po:02} "
343
344    self.__symbol_grid.lattice.print(print_function, "    ")

147    allow_loops (bool): If True, finds paths that are loops. Defaults to True.
148  """

156      allow_loops: bool = True,

180  def __add_path_edge_constraints(self):
181    solver = self.__symbol_grid.solver
182    sym: PathSymbolSet = cast(PathSymbolSet, self.__symbol_grid.symbol_set)
183
184    for p, cell in self.__symbol_grid.grid.items():
185      for d in self.__symbol_grid.lattice.edge_sharing_directions():
186        np = p.translate(d.vector)
187        dir_syms = sym.symbols_for_direction(d)
188        ncell = self.__symbol_grid.grid.get(np, None)
189        if ncell is not None:
190          opposite_syms = sym.symbols_for_direction(
191            self.__symbol_grid.lattice.opposite_direction(d))
192          cell_points_dir = Or(*[cell == s for s in dir_syms])
193          neighbor_points_opposite = Or(*[ncell == s for s in opposite_syms])
194          solver.add(Implies(cell_points_dir, neighbor_points_opposite))
195        else:
196          for s in dir_syms:
197            solver.add(cell != s)

199  def __add_path_instance_grid_constraints(self):
200    solver = self.__symbol_grid.solver
201    sym: PathSymbolSet = cast(PathSymbolSet, self.__symbol_grid.symbol_set)
202
203    for p, pi in self.__path_instance_grid.items():
204      if self.__complete:
205        solver.add(pi >= 0)
206      else:
207        solver.add(pi >= -1)
208      solver.add(pi < len(self.__symbol_grid.grid))
209
210      cell = self.__symbol_grid.grid[p]
211      solver.add(sym.is_path(cell) == (pi != -1))
212      point_index = self.__symbol_grid.lattice.point_to_index(p)
213      assert point_index is not None
214      solver.add(
215        (self.__path_order_grid[p] == 0) ==
216        (pi == point_index)
217      )
218      for d in self.__symbol_grid.lattice.edge_sharing_directions():
219        dir_syms = sym.symbols_for_direction(d)
220        np = p.translate(d.vector)
221        ncell = self.__symbol_grid.grid.get(np, None)
222        if ncell is not None:
223          cell_points_dir = Or(*[cell == s for s in dir_syms])
224          solver.add(
225            Implies(
226              cell_points_dir,
227              pi == self.__path_instance_grid[np]
228            )
229          )

231  def __all_direction_pairs(self) -> Iterable[Tuple[int, Direction, Direction]]:
232    dirs = self.__symbol_grid.lattice.edge_sharing_directions()
233    for idx, (di, dj) in enumerate(itertools.combinations(dirs, 2)):
234      yield (idx, di, dj)

236  def __add_path_order_grid_constraints(self):
237    solver = self.__symbol_grid.solver
238    sym: PathSymbolSet = cast(PathSymbolSet, self.__symbol_grid.symbol_set)
239
240    for p, po in self.__path_order_grid.items():
241      if self.__complete:
242        solver.add(po >= 0)
243      else:
244        solver.add(po >= -1)
245
246      cell = self.__symbol_grid.grid[p]
247      solver.add(sym.is_path(cell) == (po != -1))
248
249      for d in self.__symbol_grid.lattice.edge_sharing_directions():
250        try:
251          s = sym.terminal_for_direction(d)
252        except KeyError:
253          continue
254        np = p.translate(d.vector)
255        if np in self.__path_order_grid:
256          solver.add(Implies(
257            cell == s,
258            Or(
259              And(
260                self.__path_order_grid[p] == 0,
261                self.__path_order_grid[np] == 1
262              ),
263              And(
264                self.__path_order_grid[p] > 0,
265                self.__path_order_grid[np] == self.__path_order_grid[p] - 1
266              )
267            )
268          ))
269
270      for idx, d1, d2 in self.__all_direction_pairs():
271        pi = p.translate(d1.vector)
272        pj = p.translate(d2.vector)
273        if pi in self.__path_order_grid and pj in self.__path_order_grid:
274          solver.add(Implies(
275            cell == idx,
276            self.__path_order_grid[pi] != self.__path_order_grid[pj]
277          ))
278          solver.add(Implies(
279            And(cell == idx, po > 0),
280            Or(
281              And(
282                self.__path_order_grid[pi] == po - 1,
283                Or(
284                  self.__path_order_grid[pj] == po + 1,
285                  self.__path_order_grid[pj] == 0 if self.__allow_loops else BoolVal(False)
286                )
287              ),
288              And(
289                Or(
290                  self.__path_order_grid[pi] == po + 1,
291                  self.__path_order_grid[pi] == 0 if self.__allow_loops else BoolVal(False)
292                ),
293                self.__path_order_grid[pj] == po - 1
294              ),
295            )
296          ))

298  def __add_allow_terminated_paths_constraints(self):
299    if not self.__allow_terminated_paths:
300      for cell in self.__symbol_grid.grid.values():
301        self.__symbol_grid.solver.add(
302          Not(self.__symbol_grid.symbol_set.is_terminal(cell)))

309        If(self.__path_order_grid[p] == 0, 1, 0)
