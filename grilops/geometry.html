239  def __print_points(
240      self,
241      hook_function: Callable[[Point], Optional[str]],
242      ps: Iterable[Point],
243      blank: str = " ",
244      stream: IO[str] = sys.stdout):
245    """Prints something for each of the given points.
246
247    :param hook_function: A function implementing per-location display
248      behavior. It will be called for each `Point` in the lattice. If the
249      returned string has embedded newlines, it will be treated as a multi-line
250      element.  For best results, all elements should have the same number of
251      lines as each other and as blank (below).
252    :param ps: The `Point`s to print something for.
253    :param blank: What to print for `Point`s not in the lattice, or for when
254      the hook function returns None. Defaults to one space.  If it has
255      embedded newlines, it will be treated as a multi-line element.
256    :param stream: The stream to which to print the output. Defaults to
257      standard output.
258    """
259    columns = []
260    for p in ps:
261      output = None
262      if self.point_to_index(p) is not None:
263        output = hook_function(p)
264      if output is None:
265        output = blank
266      columns.append(output.split("\n"))
267    for row in zip(*columns):
268      for col in row:
269        stream.write(col)
270      stream.write("\n")

272  def print(
273      self, hook_function: Callable[[Point], Optional[str]],
274      blank: str = " ", stream: IO[str] = sys.stdout):
275    """Prints something for each space in the lattice.
276
277    Printing is done from top to bottom and left to right.
278
279    :param hook_function: A function implementing per-location display
280      behavior. It will be called for each `Point` in the lattice. If the
281      returned string has embedded newlines, it will be treated as a multi-line
282      element.  For best results, all elements should have the same number of
283      lines as each other and as blank (below).
284    :param blank: What to print for `Point`s not in the lattice, or for when
285      the hook function returns None. Defaults to one space.  If it has
286      embedded newlines, it will be treated as a multi-line element.
287    :param stream: The stream to which to print the output. Defaults to
288      standard output.
289    """
290    ps = self.points
291    min_y = ps[0].y
292    max_y = ps[-1].y
293    min_x = min(p.x for p in ps)
294    max_x = max(p.x for p in ps)
295    for y in range(min_y, max_y + 1):
296      self.__print_points(
297          hook_function,
298          (Point(y, x) for x in range(min_x, max_x + 1)),
299          blank, stream
300      )

543  def label_for_direction_pair(self, d1: Direction, d2: Direction) -> str:
544    """Returns the label for a pair of edge-sharing directions.
545
546    :param d1: The first direction.
547    :param d2: The second direction.
548
549    :return: A label representing both directions.
550
551    :raises ValueError: If there's no character defined for the direction pair.
552    """
553    ds = {d1.name, d2.name}
554
555    def char_for_pos(dirs, chars):
556      for d, c in zip(dirs, chars):
557        if d in ds:
558          ds.remove(d)
559          return chr(c)
560      return " "
561
562    ul = char_for_pos(("NW", "N", "W"), (0x2572, 0x2595, 0x2581))
563    ur = char_for_pos(("NE", "N", "E"), (0x2571, 0x258F, 0x2581))
564    ll = char_for_pos(("SW", "S", "W"), (0x2571, 0x2595, 0x2594))
565    lr = char_for_pos(("SE", "S", "E"), (0x2572, 0x258F, 0x2594))
566    return ul + ur + "\n" + ll + lr

595  def transformation_functions(
596      self,
597      allow_rotations: bool,
598      allow_reflections: bool
599      ) -> List[Callable[[Vector], Vector]]:
600    """Returns a list of `Vector` transformations.
601
602    Each returned transformation is a function that transforms a
603    `Vector` into a `Vector`. The returned list always contains at least
604    one transformation: the identity function.  The transformations
605    returned are all transformations satisfying the given constraints.
606
607    :param allow_rotations: Whether rotation is an allowed transformation.
608    :param allow_reflections: Whether reflection is an allowed transformation.
609
610    :return: A list of `Vector` transformation functions.
611    """
612    if allow_rotations:
613      if allow_reflections:
614        return [
615            lambda v: v,                                                    # Identity
616            lambda v: Vector((v.dy + 3 * v.dx) // 2, (-v.dy + v.dx) // 2),  # Rotate 60 deg
617            lambda v: Vector((-v.dy + 3 * v.dx) // 2, (-v.dy - v.dx) // 2), # Rotate 120 deg
618            lambda v: Vector(-v.dy, -v.dx),                                 # Rotate 180 deg
619            lambda v: Vector((-v.dy - 3 * v.dx) // 2, (v.dy - v.dx) // 2),  # Rotate 240 deg
620            lambda v: Vector((v.dy - 3 * v.dx) // 2, (v.dy + v.dx) // 2),   # Rotate 300 deg
621            lambda v: Vector(-v.dy, v.dx),                                  # Reflect across 0 deg
